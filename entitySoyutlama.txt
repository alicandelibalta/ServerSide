// models/User.js

class User {
  constructor(id, username, email, password) {
    this.id = id;
    this.username = username;
    this.email = email;
    this.password = password;
  }
}

function createUser(id, username, email, password) {
  return new User(id, username, email, password);
}

module.exports = { createUser };
